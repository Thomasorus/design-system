    :root {
    /* padding y for input */
    /* 3 is min, 5 is max */
    --padding-input-y: calc(((3 / 16) * 1rem) + (5 - 3) * var(--fluid-bp));

     /* padding x for input */
    /* 9 is min, 17 is max */
     --padding-input-x: calc(((9 / 16) * 1rem) + (17 - 9) * var(--fluid-bp));
}
